-- Create the database
CREATE DATABASE IF NOT EXISTS music_db;
USE music_db;

-- Create Artist table 
CREATE TABLE Artist (
    artist_id INT,
    name VARCHAR(100)
);

-- Create Album table  
CREATE TABLE Album (
    album_id INT,
    title VARCHAR(100)
);

-- Create Track table  
CREATE TABLE Track (
    track_id INT,
    title VARCHAR(100),
    album_id INT,
    artist_id INT
);
